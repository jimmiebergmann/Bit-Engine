Bit::Matrix4x4 ProjectionMatrix;
ProjectionMatrix.Orthographic( 0.0f, 800.0f, 0.0f, 600.0f, -1.0f, 1.0f );

// Load the matrix manager
// Bit::RenderManager::Load( pGraphicDevice ); // Use standard shaders
Bit::RenderManager::Load( pGraphicDevice, pCustomShaderProgram );

// Bind the matrix manager
Bit::RenderManager::Bind( );

// Set the matrices
Bit::RenderManager::MatrixModeProjection( );
Bit::RenderManager::SetMatrix( ProjectionMatrix );
Bit::RenderManager::MatrixModeModelView( );
Bit::RenderManager::SetIdentity( );

// Change the standard shader attributes
Bit::RenderManager::SetShaderPositionAttribute( "Position" );
Bit::RenderManager::SetShaderTextureCoordinateAttribute( "TextureCoordinate" );
Bit::RenderManager::SetShaderTextureAttribute( "Texture" );
Bit::RenderManager::SetShaderColorAttribute( "Color" );

// Push the current matrix
Bit::RenderManager::PushMatrix( );

// Translate and rotate
Bit::RenderManager::Rotat( 90.0f, 0.0f, 0.0f, 1.0f );
Bit::RenderManager::Translate( 100.0f, 50.0f, 0.0f );

// Render
// ...

// Pop the current matrix
Bit::RenderManager::Pop( );


// Unbind the matrix manager
Bit::MatrixManager::Unbind( );
